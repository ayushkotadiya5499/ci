name: CI/CD for Blue-Green

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  build-and-test:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Run DVC pipeline
        run: |
          dvc repro

      - name: Run tests
        run: |
          pip install pytest
          pytest -q

      # ૧. ECR માં લોગિન કરો
      - name: Login to AWS ECR
        if: success()
        run: |
          aws configure set aws_access_key_id ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws configure set aws_secret_access_key ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws ecr get-login-password --region us-east-1 | docker login --username AWS --password-stdin 982534348859.dkr.ecr.us-east-1.amazonaws.com

      # ૨. ડોકર ઇમેજ બિલ્ડ કરો
      - name: Build Docker image
        if: success()
        run: |
          docker build -t salary_ecr .

      # ૩. ઇમેજને ટેગ કરો
      - name: Tag Docker image
        if: success()
        run: |
          docker tag salary_ecr:latest 982534348859.dkr.ecr.us-east-1.amazonaws.com/salary_ecr:latest

      # ૪. ઇમેજને ECR માં પુશ કરો
      - name: Push Docker image to AWS ECR
        if: success()
        run: |
          docker push 982534348859.dkr.ecr.us-east-1.amazonaws.com/salary_ecr:latest

      # ૫. ડિપ્લોયમેન્ટ માટે ફાઈલોને ઝિપ કરો
      - name: Zip files for deployment
        if: success()
        run: |
          zip -r deployment.zip appspec.yml deploy/scripts/install_dependencies.sh deploy/scripts/start_docker.sh

      # ૬. ઝિપ ફાઈલને S3 બકેટમાં અપલોડ કરો
      - name: Upload ZIP to S3
        if: success()
        run: |
          aws s3 cp deployment.zip s3://ayush-code-deploy-15/deployment.zip

      # ૭. Blue/Green Deployment શરૂ કરો
      - name: Deploy to AWS CodeDeploy
        if: success()
        run: |
          aws deploy create-deployment \
            --application-name ayush-deploy-15 \
            --deployment-group-name ayush-blue-green-15 \
            --deployment-config-name CodeDeployDefault.AllAtOnce \
            --s3-location bucket=ayush-code-deploy-15,key=deployment.zip,bundleType=zip \
            --file-exists-behavior OVERWRITE \
            --region us-east-1
