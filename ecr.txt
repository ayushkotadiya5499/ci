name: CI

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  build-and-test:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Run DVC pipeline
        run: |
          dvc repro

      - name: Run tests
        run: |
          pip install pytest
          pytest -q

      - name: Login to AWS ECR
        if: success()
        run: |
          aws configure set aws_access_key_id ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws configure set aws_secret_access_key ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws ecr get-login-password --region us-east-1 | docker login --username AWS --password-stdin 982534348859.dkr.ecr.us-east-1.amazonaws.com

      - name: Build Docker image
        if: success()
        run: |
          docker build -t salary_ecr .

      - name: Tag Docker image
        if: success()
        run: |
          docker tag salary_ecr:latest 982534348859.dkr.ecr.us-east-1.amazonaws.com/salary_ecr:latest

      - name: Push Docker image to AWS ECR
        if: success()
        run: |
          docker push 982534348859.dkr.ecr.us-east-1.amazonaws.com/salary_ecr:latest
